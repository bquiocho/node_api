# How to create a REST API

Install node js

# Create a git/github repository

# Open CLI terminal
# cd into that folder
cd ~/Projects/development/API

# Get the latest stable release of Sails
npm install sails -g

# Create the sails app
sails new jats_api

# cd into the new folder
cd jats_api

# Fire up the server (Lift the sails)
sails lift
# Open up localhost/1337 to view your sails website

# Generating a REST API.
sails generate api user
sails generate api action
# This will create two files: a model and a controller.
